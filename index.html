    Supabase ranking setup memo

    Enter SUPABASE_URL and SUPABASE_ANON_KEY near the top of this code to enable the shared ranking. Never put a service_role key in public HTML. 

    create table public.block_drop_scores (
  id bigint generated by default as identity primary key,
  player_name text not null,
  score integer not null,
  lines integer not null default 0,
  level integer not null default 1,
  created_at timestamptz not null default now()
);

alter table public.block_drop_scores enable row level security;

create policy "Scores are readable by everyone"
on public.block_drop_scores
for select
to anon
using (true);

create policy "Anyone can insert scores"
on public.block_drop_scores
for insert
to anon
with check (
  char_length(player_name) between 1 and 16
  and score >= 0
  and lines >= 0
  and level >= 1
);
